860 private static void actualitzarFlux(List<Node> cami, Map<Node, Map<Node, Float>> fluxSortides, Map<Node, Map<Node, Float>> fluxEntrades) {
861 // Trobar el flux mínim en el camí
862 float fluxMinim = Float.MAX_VALUE;
863 for (int i = 0; i < cami.size() - 1; i++) {
864 Node nodeActual = cami.get(i);
865 Node nodeSeguent = cami.get(i + 1);
866 float flux;
867 if(fluxSortides.get(nodeActual).containsKey(nodeSeguent)) flux = fluxSortides.get(nodeActual).get(nodeSeguent);
868 else flux = fluxEntrades.get(nodeActual).get(nodeSeguent);
869
870 if (flux < fluxMinim) fluxMinim = flux;
871 }
872
873 // Actualitzar fluxSortides i fluxEntrades
874 for (int i = 0; i < cami.size() - 1; i++) {
875 Node nodeActual = cami.get(i);
876 Node nodeSeguent = cami.get(i + 1);
877
878 if (fluxSortides.get(nodeActual).containsKey(nodeSeguent)) {
879 // El nodeSeguent és una sortida del nodeActual
880 float fluxSortida = fluxSortides.get(nodeActual).get(nodeSeguent);
881 fluxSortides.get(nodeActual).put(nodeSeguent, fluxSortida - fluxMinim);
882 float fluxEntrada = fluxEntrades.get(nodeSeguent).get(nodeActual);
883 fluxEntrades.get(nodeSeguent).put(nodeActual, fluxEntrada + fluxMinim);
884 }
885 else if (fluxEntrades.get(nodeActual).containsKey(nodeSeguent)) {
886 // El nodeSeguent és una entrada del nodeActual
887 float fluxEntrada = fluxEntrades.get(nodeActual).get(nodeSeguent);
888 fluxEntrades.get(nodeActual).put(nodeSeguent, fluxEntrada - fluxMinim);
889 float fluxSortida = fluxSortides.get(nodeSeguent).get(nodeActual);
890 fluxSortides.get(nodeSeguent).put(nodeActual, fluxSortida + fluxMinim);
891 }
892 }
