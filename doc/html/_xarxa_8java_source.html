415 public void dfs(Node node, float demandaPrimera, Map<Canonada, Float> canonades) {
416 Iterator<Canonada> it = entrades(node);
417 int count = 0;
418 float total = 0;
419 List<Float> capacidades = new ArrayList<>();
420
421 while(it.hasNext()) {
422 Canonada canonada = it.next();
423 capacidades.add(canonada.capacitat());
424 total += canonada.capacitat();
425 count++;
426 }
427 boolean capacidadesIguales = new HashSet<>(capacidades).size() == 1;
428 it = entrades(node);
429 if (demandaPrimera != 0 && count > 0) {
430 while (it.hasNext()) {
431 Canonada canonada = it.next();
432 Node nodeReceptor = canonada.node1();
433 demandas.putIfAbsent(nodeReceptor, 0f);
434 if (!canonada.node1().aixetaOberta()) {
435 dfs(nodeReceptor, 0, canonades);
436 } else {
437 float demandaNova;
438 if (count > 1 && capacidadesIguales) {
439 demandaNova = demandaPrimera / count;
440 } else {
441 demandaNova = (demandaPrimera / total) * canonada.capacitat();
442 }
443 calculadoraDemanda(nodeReceptor, demandaNova, canonada,canonades);
444 }
445 }

460 public void calculadoraDemanda(Node nodeReceptor, float demandaNova, Canonada canonada,Map<Canonada, Float> canonades){
461 float capacidad = canonada.capacitat();
462 float demandaAjustada = Math.min(demandaNova, capacidad);
463 float demandaActual = demandas.get(nodeReceptor);
464 float demandaFinal = demandaActual + demandaAjustada;
465 demandas.put(nodeReceptor, demandaFinal);
466 float demandaCanonada = canonades.get(canonada);
467 float demandaCanonadaNova = demandaCanonada + demandaAjustada;
468 if (capacidad == demandaCanonada) {
469 demandas.put(nodeReceptor, demandaActual);
470 demandaAjustada = 0;
471 } else {
472 if (capacidad == demandaCanonadaNova) {
473 canonades.put(canonada, demandaCanonadaNova);
474 } else if (capacidad < demandaCanonadaNova) {
475 canonades.put(canonada, demandaAjustada + demandaCanonada);
476 demandas.put(nodeReceptor, (demandaAjustada - demandaCanonada) + demandaActual);
477 demandaAjustada = 0;
478 } else {
479 canonades.put(canonada, demandaCanonadaNova);
480 }
481 }
482 dfs(nodeReceptor, demandaAjustada, canonades);

492 public float cabal(Node node){
493 //Pre: node pertany a la xarxa

531 public void dfsCabals(Node node, float cabalOrigen){
532 Iterator <Canonada> it = sortides(node);
533 if (cabalOrigen != 0){
534 while(it.hasNext()) {
535 Canonada canonada = it.next();
536 Node nodeReceptor = canonada.node2();
537 cabals.putIfAbsent(nodeReceptor, 0f);
538 if (!canonada.node2().aixetaOberta()) {
539 dfs(nodeReceptor, 0, canonadaCabals);
540 } else {
541 float diferencia = cabalOrigen / demanda(node);
542 float nouCabalCanonada = diferencia * canonades.get(canonada);
543 float cabalCanonada = canonadaCabals.get(canonada);
544 canonadaCabals.put(canonada, nouCabalCanonada + cabalCanonada);
545 float cabalActual = cabals.get(nodeReceptor);
546 cabals.put(nodeReceptor, cabalActual + nouCabalCanonada);
547 dfsCabals(nodeReceptor, nouCabalCanonada);

605 private void dfs(Node actual, Set<Node> visitados) {
606 visitados.add(actual);
607 if (actual != null) {
608 Iterator<Edge> salientes = graph.getNode(actual.id()).leavingEdges().iterator();
609 while (salientes.hasNext()) {
610 Edge edge = salientes.next();
611 String nom = edge.getOpposite(graph.getNode(actual.id())).getId();
612 Node destino = getNodeNou(nom);
613 if (destino != null && !visitados.contains(destino)) {
614 dfs(destino, visitados);
615 }
616 }
617 }
618 if(actual != null) {
619 Iterator<Edge> entrantes = graph.getNode(actual.id()).enteringEdges().iterator();
620 while (entrantes.hasNext()) {
621 Edge edge = entrantes.next();
622 String nom = edge.getOpposite(graph.getNode(actual.id())).getId();
623 Node origen = getNodeNou(nom);
624 if (origen != null && !visitados.contains(origen)) {
625 dfs(origen, visitados);
