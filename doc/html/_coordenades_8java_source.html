73 public double distancia(Coordenades c) {
74 //Pre: ---
75 //Post: Retorna la distància entre aquestes coordenades i c, expressada en km
76 double radiTerra = 6371.0; // Radi de la Terra en quilòmetres
77 double dLat = Math.toRadians(c.latitud - this.latitud);
78 double dLon = Math.toRadians(c.longitud - this.longitud);
79 double lat1 = Math.toRadians(this.latitud);
80 double lat2 = Math.toRadians(c.latitud);
81
82 double a = Math.sin(dLat / 2) * Math.sin(dLat / 2) + Math.sin(dLon / 2) * Math.sin(dLon / 2) * Math.cos(lat1) * Math.cos(lat2);
83 double cHav = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
84
85 return radiTerra * cHav;
86 }
